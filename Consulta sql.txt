CREATE TABLE CRM (
    id_cliente INT NOT NULL AUTO_INCREMENT,
    nombre VARCHAR(100) NOT NULL,
    apellidos varchar(150),
    direccion varchar(150),
    entre_calles varchar(200),
    correo VARCHAR(100),
    numero_telefonico BIGINT,
    activo BOOLEAN DEFAULT TRUE NOT NULL,
    fecha_inicio TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_finalizacion TIMESTAMP,
    PRIMARY KEY (id_cliente)
);

CREATE TABLE RH (
    id_trabajador INT NOT NULL AUTO_INCREMENT,
    nombre VARCHAR(100) NOT NULL,
    apellidos varchar(150),
    rol BOOLEAN DEFAULT FALSE NOT NULL,
    password VARCHAR(150) NULL,
    edad integer,
    direccion varchar(150),
    entre_calles varchar(200),
    correo VARCHAR(100),
    numero_telefonico BIGINT,
    activo BOOLEAN DEFAULT TRUE NOT NULL,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id_trabajador)
);

CREATE TABLE productos (
    id_producto INT NOT NULL AUTO_INCREMENT,
    nombre VARCHAR(100) NOT NULL,
    cantidad double,
    reserva double null,
    medida varchar(120),
    conservado varchar(120),
    tipo varchar(120),
    marca varchar(120),
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id_producto)
);

CREATE TABLE material (
    id_material INT NOT NULL AUTO_INCREMENT,
    nombre VARCHAR(100) NOT NULL,
    cantidad double,
    marca varchar(120),
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id_material)
);

CREATE TABLE grupos_trabajadores (
    id_grupo_trabajadores VARCHAR(120) PRIMARY KEY,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE grupos_productos (
    id_grupo_productos VARCHAR(120) PRIMARY KEY,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE grupos_materiales (
    id_grupo_materiales VARCHAR(120) PRIMARY KEY,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE grupo_trabajadores (
    id_grupo_trabajadores VARCHAR(120),
    id_trabajador INT,

    PRIMARY KEY (id_grupo_trabajadores, id_trabajador),

    CONSTRAINT fk_grupotrabajadores_grupostrabajadores
    FOREIGN KEY (id_grupo_trabajadores)
    REFERENCES grupos_trabajadores(id_grupo_trabajadores),

    CONSTRAINT fk_RH_grupotrabajadores
    FOREIGN KEY (id_trabajador)
    REFERENCES rh(id_trabajador)
);

CREATE TABLE grupo_productos (
    id_grupo_productos VARCHAR(120),
    id_producto INT,
    cantidad double,
    
    PRIMARY KEY (id_grupo_productos, id_producto),

    CONSTRAINT fk_grupoproductos_gruposproductos
    FOREIGN KEY (id_grupo_productos)
    REFERENCES grupos_productos(id_grupo_productos),

    CONSTRAINT fk_productos_grupoproductos
    FOREIGN KEY (id_producto)
    REFERENCES productos(id_producto)
);

CREATE TABLE grupo_materiales (
    id_grupo_materiales VARCHAR(120),
    id_material INT,
    cantidad INTEGER,
    
    PRIMARY KEY (id_grupo_materiales, id_material),

    CONSTRAINT fk_grupomateriales_gruposmateriales
    FOREIGN KEY (id_grupo_materiales)
    REFERENCES grupos_materiales(id_grupo_materiales),

    CONSTRAINT fk_materiales_grupomateriales
    FOREIGN KEY (id_material)
    REFERENCES material(id_material)
);

CREATE TABLE operacion (
    id_operacion INT NOT NULL AUTO_INCREMENT,
    trabajo_realizado varchar(200),
    id_grupo_trabajadores VARCHAR(120) NOT NULL,
    id_grupo_productos VARCHAR(120) NOT NULL,
    id_grupo_materiales VARCHAR(120) NOT NULL,
    id_cliente INT NOT NULL,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_finalizacion TIMESTAMP,
    PRIMARY KEY (id_operacion),
    
    CONSTRAINT fk_grupostrabajadores_operacion
    FOREIGN KEY (id_grupo_trabajadores)
    REFERENCES grupos_trabajadores(id_grupo_trabajadores),
    
    CONSTRAINT fk_gruposproductos_operacion
    FOREIGN KEY (id_grupo_productos)
    REFERENCES grupos_productos(id_grupo_productos),
    
    CONSTRAINT fk_gruposmateriales_operacion
    FOREIGN KEY (id_grupo_materiales)
    REFERENCES grupos_materiales(id_grupo_materiales),
    
    CONSTRAINT fk_cliente_operacion
    FOREIGN KEY (id_cliente)
    REFERENCES CRM(id_cliente)
);

CREATE TABLE notas (
    id_notas INT NOT NULL AUTO_INCREMENT,
    escrito VARCHAR(500),
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id_notas)
);

CREATE TRIGGER `before_insert_grupo_productos` 
BEFORE INSERT ON `grupo_productos` 
FOR EACH ROW 
BEGIN
    DECLARE v_total_antiguo DOUBLE;
    DECLARE v_reserva DOUBLE;
    DECLARE v_nuevo_total DOUBLE;
    DECLARE v_nueva_cantidad_envases INT;
    DECLARE v_cant_antigua_envases INT;

    -- 1. Obtener datos actuales del producto
    SELECT total, reserva, cantidad INTO v_total_antiguo, v_reserva, v_cant_antigua_envases
    FROM productos
    WHERE id_producto = NEW.id_producto;

    -- 2. Restar lo consumido al total general
    SET v_nuevo_total = v_total_antiguo - NEW.consumido;

    -- 3. Calcular nueva cantidad de envases (Redondeo hacia arriba)
    -- Ejemplo: 2900 / 600 = 4.83 -> CEIL(4.83) = 5
    SET v_nueva_cantidad_envases = CEIL(v_nuevo_total / v_reserva);

    -- 4. Calcular la 'cantidad usada' (la diferencia de envases físicos)
    -- Ejemplo: 7 envases iniciales - 5 finales = 2 usados
    SET NEW.cantidad_usada = v_cant_antigua_envases - v_nueva_cantidad_envases;

    -- 5. Actualizar la tabla Maestra (productos)
    UPDATE productos 
    SET total = v_nuevo_total,
        cantidad = v_nueva_cantidad_envases
    WHERE id_producto = NEW.id_producto;

END$$
